def initialize(username, api_key)
  @username = username
  @api_key = api_key
  @stub_responses = nil

  result = exec_request(build_payload({:username => @username}), 'http://api.nutshell.com/v1/json')
  api_host = result['api']
  @api_url = "https://#{api_host}/api/v1/json"
end
